SELECT
	TABLE_NAME AS "Name",
	CASE
		WHEN (TABLE_TYPE = 'VIEW') THEN
			CAST(1 AS unsigned)
		ELSE
			CAST(0 AS unsigned)
		END AS "isVT"
FROM INFORMATION_SCHEMA.TABLES
WHERE
	(
		(TABLE_TYPE = 'BASE TABLE')
		OR
		(TABLE_TYPE = 'VIEW')
	)
	AND
	(TABLE_SCHEMA != 'mysql')
ORDER BY "Name"
