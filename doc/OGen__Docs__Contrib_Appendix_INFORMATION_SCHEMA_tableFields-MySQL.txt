SELECT
	_t1.table_name,
	_t1.column_name,
	CASE
		WHEN _t1.is_nullable = 'No' THEN
			CAST(0 AS unsigned)
		ELSE
			CAST(1 AS unsigned)
		END AS "is_nullable",
	_t1.data_type,
	_t1.character_maximum_length,
	CASE
		WHEN ((t6.table_type != 'VIEW') AND (_t1.column_key = 'PRI')) THEN
			CAST(1 AS unsigned)
		ELSE
			CAST(0 AS unsigned)
	END AS "is_pk",
	CASE
		WHEN ((t6.table_type != 'VIEW') AND (_t1.extra = 'auto_increment')) THEN
			CAST(1 AS unsigned)
		ELSE
			CAST(0 AS unsigned)
	END AS "is_identity",
	NULL AS "fk_table_name",
	NULL AS "fk_field_name", 
	_t1.column_default, 
	_t1.collation_name, 
	_t1.numeric_precision, 
	_t1.numeric_scale
FROM information_schema.columns _t1
	LEFT JOIN information_schema.tables t6 ON (
		(t6.table_schema = _t1.table_schema) 
		AND 
		(t6.table_name = _t1.table_name)
	)
WHERE
	(_t1.table_schema = 'OGen-NTier_UTs')
	AND
	(
		(_t1.table_name = 'User')
		OR
		(_t1.table_name = 'UserGroup')
	)
ORDER BY
	_t1.table_name, 
	_t1.ordinal_position
